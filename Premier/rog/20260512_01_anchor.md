## 概要

選択クリップのアンカーポイントを 3×3 グリッドのボタン1クリックで移動する。  
Crop / Alpha / Mask / Transform エフェクトの範囲を考慮したスナップが可能。

## 重要な技術知見

### アンカー / 位置の単位系
```
anchorProp.getValue()  → 正規化 [0,1]（UIではピクセル表示）
anchorProp.setValue()  → 正規化 [0,1] を渡す
posProp.getValue()     → 正規化 [0,1]（UIではピクセル表示）
posProp.setValue()     → 正規化 [0,1] を渡す
```
UI の「位置: 960, 540」は正規化値 [0.5, 0.5] をピクセル表示したもの。

### Hold Position の正しい計算式（参照スクリプト準拠）
```javascript
// ① アンカーデルタ(正規化) × ソース寸法 × スケール → ソースピクセル変位
var pixelMoveX = (newAX - curAnchor[0]) * sw * (curScale / 100);
var pixelMoveY = (newAY - curAnchor[1]) * sh * (curScale / 100);
// ② ÷ シーケンス寸法 → 位置の正規化デルタ
posProp.setValue([
    curPos[0] + pixelMoveX / seqW,
    curPos[1] + pixelMoveY / seqH
], 1);
```
**罠:** `/seqW` `/seqH` の除算を省くと位置が ±32767 (INT16_MAX) にクランプされる。

### ソースサイズ取得の注意点
- `_sourceSize()` はメタデータの正規表現で解像度を取得
- `\d{3,5}` だと `65534`（0xFFFE、センチネル値）を誤検出する
- 修正: `\d{3,4}` に変更（最大 9999px）＋ `> 16384` のサニティチェック

### キーフレーム API（正しい使い方）
```javascript
// NG: numKeys / getKeyTime(index) は存在しない
// OK:
var keys = posProp.getKeys();           // Time オブジェクトの配列を返す
var val  = posProp.getValueAtKey(kTime);
posProp.setValueAtKey(kTime, newVal, 1);
```

### コンポーネント名（日英両対応必須）
| 処理対象 | displayName |
|----------|-------------|
| ベクトルモーション | `Motion` / `モーション` |
| Transform エフェクト | `Transform` / `トランスフォーム` |
| Crop エフェクト | `Crop` / `クロップ` / `切り抜き` |
| 不透明度 | `Opacity` / `不透明度` |

### マスク検出（API 制限・確定）
- `clip.components[n].properties` にマスクグループは公開されない
- `comp.numMasks` / `comp.getMask` も存在しない
- **結論: Premiere CEP API ではマスクのパス情報は取得不可能**

### フレームエクスポート（API 制限・確定）
- `seq.exportFramePNG()` → このバージョンに存在しない
- QE API (`qe.project.getActiveSequence().exportStill()` 等) → 呼び出し失敗
- **結論: 手動入力モードにフォールバックが現実的**

## 未解決 / 今後の課題

1. **Hold Position + キーフレーム**  
   キーフレームがある場合は現在警告のみ。全キーフレームをシフトするには
   `getValueAtKey(Time)` + `setValueAtKey(Time, val, 1)` の反復が必要だが、
   試みたところ正しく動作しなかった（単位系の問題と思われる）。

2. **Transform エフェクトの座標系確認**  
   Motion と同じ正規化 [0,1] を使うと仮定して実装。未テスト。

3. **マスク自動検出**  
   現状 API では不可。将来的に Premiere が API を公開した場合に対応可能な
   設計にしてある（`_detectMask` 関数は残置）。

4. **ソースサイズ取得の信頼性**  
   テキスト/MOGRT クリップ等でメタデータがない場合はシーケンスサイズをフォールバック。

```javascript
// アンカー CEP 拡張 - hostscript.jsx  (v2 / 2026-05-12)
// 参照: W:\共有ドライブ\wc動画\sync\Assets\ymst_jsx\Premier\アンカーポイント.jsx

// ── Public API ───────────────────────────────────────────────────────────────

function AS_getClipInfo() {
    try {
        var seq = app.project.activeSequence;
        if (!seq) return JSON.stringify({ success: false, error: 'アクティブなシーケンスがありません' });

        var clip = _firstSelected(seq);
        if (!clip) return JSON.stringify({ success: false, error: 'クリップを選択してください' });

        var mediaPath = '';
        var sw = 0, sh = 0;
        if (clip.projectItem) {
            try { mediaPath = clip.projectItem.getMediaPath() || ''; } catch (e) {}
            var sz = _sourceSize(clip.projectItem);
            sw = sz[0]; sh = sz[1];
        }
        if (!sw || !sh) { sw = seq.frameSizeHorizontal || 1920; sh = seq.frameSizeVertical || 1080; }

        var ext = mediaPath.toLowerCase().replace(/.*\./, '');
        var supportsAlpha = /^(png|tif|tiff|psd|exr|psb)$/.test(ext);

        var transforms = [];
        for (var tci = 0; tci < clip.components.numItems; tci++) {
            var tcdn = clip.components[tci].displayName || '';
            if (/^(Transform|トランスフォーム)$/.test(tcdn)) {
                transforms.push({ index: tci, displayName: tcdn });
            }
        }

        return JSON.stringify({
            success: true, name: clip.name || '', mediaPath: mediaPath,
            supportsAlpha: supportsAlpha, sourceWidth: sw, sourceHeight: sh,
            crop: _detectCrop(clip), mask: null, transforms: transforms
        });
    } catch (e) { return JSON.stringify({ success: false, error: e.toString() }); }
}

function AS_getAlphaSourcePath() {
    try {
        var seq = app.project.activeSequence;
        if (!seq) return JSON.stringify({ success: false, error: 'アクティブなシーケンスがありません' });
        for (var i = 0; i < seq.videoTracks.numTracks; i++) {
            var track = seq.videoTracks[i];
            for (var j = 0; j < track.clips.numItems; j++) {
                var clip = track.clips[j];
                if (!clip.isSelected() || !clip.projectItem) { continue; }
                var path = '';
                try { path = clip.projectItem.getMediaPath() || ''; } catch (e) {}
                if (!path) { continue; }
                var ext = path.toLowerCase().replace(/.*\./, '');
                if (/^(png|tif|tiff|psd|exr|psb)$/.test(ext)) {
                    return JSON.stringify({ success: true, path: path });
                }
            }
        }
        return JSON.stringify({ success: false, error: 'このクリップはアルファ非対応です (PNG/TIF/PSD/EXR のみ)' });
    } catch (e) { return JSON.stringify({ success: false, error: e.toString() }); }
}

function AS_anchorSnap(xMult, yMult, holdPos, alphaBoundsJson, useCrop, maskBoundsJson, transformIndicesJson) {
    try {
        var x = parseFloat(xMult), y = parseFloat(yMult);
        var hold = (holdPos === 'true'), doCrop = (useCrop === 'true');

        var alphaBounds = null;
        if (alphaBoundsJson && alphaBoundsJson !== 'null') {
            try { alphaBounds = JSON.parse(alphaBoundsJson); } catch (e) {}
        }
        var maskBounds = null;
        if (maskBoundsJson && maskBoundsJson !== 'null') {
            try { maskBounds = JSON.parse(maskBoundsJson); } catch (e) {}
        }
        var transformIndices = [];
        if (transformIndicesJson && transformIndicesJson !== 'null' && transformIndicesJson !== '[]') {
            try { transformIndices = JSON.parse(transformIndicesJson); } catch (e) {}
        }

        var seq = app.project.activeSequence;
        if (!seq) return JSON.stringify({ success: false, error: 'アクティブなシーケンスがありません' });

        var seqW = seq.frameSizeHorizontal || 1920;
        var seqH = seq.frameSizeVertical   || 1080;
        var count = 0, fallback = false, keyFrameWarning = false;

        for (var i = 0; i < seq.videoTracks.numTracks; i++) {
            var track = seq.videoTracks[i];
            for (var j = 0; j < track.clips.numItems; j++) {
                var clip = track.clips[j];
                if (!clip.isSelected()) { continue; }

                var sw = 0, sh = 0;
                if (clip.projectItem) { var sz = _sourceSize(clip.projectItem); sw = sz[0]; sh = sz[1]; }
                if (!sw || !sh) { sw = seqW; sh = seqH; fallback = true; }
                if (sw > 16384 || sh > 16384) { sw = seqW; sh = seqH; fallback = true; }

                var bounds = { left: 0, top: 0, right: 1, bottom: 1 };
                if (alphaBounds) { bounds = _intersect(bounds, alphaBounds); }
                if (doCrop) {
                    var cr = _detectCrop(clip);
                    if (cr) {
                        bounds = _intersect(bounds, {
                            left: cr.leftPct/100, top: cr.topPct/100,
                            right: 1-cr.rightPct/100, bottom: 1-cr.bottomPct/100
                        });
                    }
                }
                if (maskBounds) { bounds = _intersect(bounds, maskBounds); }

                var newAX = bounds.left + x * (bounds.right  - bounds.left);
                var newAY = bounds.top  + y * (bounds.bottom - bounds.top);

                for (var c = 0; c < clip.components.numItems; c++) {
                    var cdn = clip.components[c].displayName || '';
                    if (cdn !== 'Motion' && cdn !== 'モーション') { continue; }
                    var r = _snapComponent(clip.components[c], newAX, newAY, sw, sh, seqW, seqH, hold);
                    if (r.snapped) { count++; }
                    if (r.keyFrameWarning) { keyFrameWarning = true; }
                }
                for (var ti = 0; ti < transformIndices.length; ti++) {
                    var idx = transformIndices[ti];
                    if (idx < 0 || idx >= clip.components.numItems) { continue; }
                    var r2 = _snapComponent(clip.components[idx], newAX, newAY, sw, sh, seqW, seqH, hold);
                    if (r2.snapped) { count++; }
                    if (r2.keyFrameWarning) { keyFrameWarning = true; }
                }
            }
        }

        if (count === 0) return JSON.stringify({ success: false, error: 'クリップが選択されていないか、Motionエフェクトが見つかりません' });
        return JSON.stringify({ success: true, count: count, fallback: fallback, keyFrameWarning: keyFrameWarning });
    } catch (e) { return JSON.stringify({ success: false, error: e.toString() }); }
}

// ── Private helpers ──────────────────────────────────────────────────────────

// anchor/pos は正規化[0,1]。Hold Position 式（参照スクリプト準拠）:
//   pixelMove = deltaAnchor(norm) × srcSize × scale/100
//   deltaPos  = pixelMove / seqSize
function _snapComponent(comp, newAX, newAY, sw, sh, seqW, seqH, hold) {
    var result = { snapped: false, keyFrameWarning: false };
    var posProp = null, scaleProp = null, anchorProp = null;
    for (var p = 0; p < comp.properties.numItems; p++) {
        var dn = comp.properties[p].displayName || '';
        if      (/^(Position|位置)$/.test(dn))                { posProp    = comp.properties[p]; }
        else if (/^(Anchor Point|アンカーポイント)$/.test(dn)) { anchorProp = comp.properties[p]; }
        else if (/^(Scale|スケール)$/.test(dn) && !scaleProp)  { scaleProp  = comp.properties[p]; }
    }
    if (!anchorProp) { return result; }

    var curAnchor = anchorProp.getValue();
    anchorProp.setValue([newAX, newAY], 1);
    result.snapped = true;

    if (hold && posProp && scaleProp) {
        var keys = null;
        try { keys = posProp.getKeys(); } catch(e) {}
        if (keys && keys.length > 0) {
            result.keyFrameWarning = true;
        } else {
            var curScale = scaleProp.getValue();
            var curPos   = posProp.getValue();
            var pixelMoveX = (newAX - curAnchor[0]) * sw * (curScale / 100);
            var pixelMoveY = (newAY - curAnchor[1]) * sh * (curScale / 100);
            posProp.setValue([curPos[0] + pixelMoveX / seqW, curPos[1] + pixelMoveY / seqH], 1);
        }
    }
    return result;
}

function _firstSelected(seq) {
    for (var i = 0; i < seq.videoTracks.numTracks; i++) {
        var track = seq.videoTracks[i];
        for (var j = 0; j < track.clips.numItems; j++) {
            if (track.clips[j].isSelected()) { return track.clips[j]; }
        }
    }
    return null;
}

// \d{3,4} で最大4桁に制限し 65534 (0xFFFE) 等の誤値を排除
function _sourceSize(projectItem) {
    var w = 0, h = 0;
    try {
        var pMeta = projectItem.getProjectMetadata() || '';
        var xMeta = projectItem.getXMPMetadata()     || '';
        var m1 = pMeta.match(/>(?:<!\[CDATA\[)?\s*(\d{3,4})\s*[xX]\s*(\d{3,4})\s*(?:\([^)]+\))?\s*(?:\]\]>)?</i);
        if (m1) { w = parseInt(m1[1], 10); h = parseInt(m1[2], 10); }
        if (!w || !h) {
            var m2 = pMeta.match(/(?:>|\[CDATA\[|\s)(\d{3,4})\s*[xX]\s*(\d{3,4})(?:<|\]\]>|\s|\()/i);
            if (m2) { w = parseInt(m2[1], 10); h = parseInt(m2[2], 10); }
        }
        if (!w || !h) {
            var wm = xMeta.match(/<tiff:ImageWidth>(\d+)/i)      || xMeta.match(/<exif:PixelXDimension>(\d+)/i) ||
                     xMeta.match(/<stDim:w>(\d+)/i)               || xMeta.match(/stDim:w="(\d+)"/i);
            var hm = xMeta.match(/<tiff:ImageLength>(\d+)/i)     || xMeta.match(/<exif:PixelYDimension>(\d+)/i) ||
                     xMeta.match(/<stDim:h>(\d+)/i)               || xMeta.match(/stDim:h="(\d+)"/i);
            if (wm && hm) { w = parseInt(wm[1], 10); h = parseInt(hm[1], 10); }
        }
    } catch (e) {}
    return [w, h];
}

function _detectCrop(clip) {
    var cropResult = null;
    try {
        for (var c = 0; c < clip.components.numItems; c++) {
            var comp = clip.components[c], cdn = comp.displayName || '';
            var L = NaN, R = NaN, T = NaN, B = NaN, hitCount = 0;
            for (var p = 0; p < comp.properties.numItems; p++) {
                var prop = comp.properties[p], pdn = prop.displayName || '', v;
                try { var raw = prop.getValue(); if (raw !== null && typeof raw !== 'object') { v = parseFloat(raw); } } catch (e) { continue; }
                if (typeof v !== 'number' || isNaN(v)) { continue; }
                if      (/^[Ll]eft$|^左$|^左端$/.test(pdn) || (/切り抜き|[Cc]rop/i.test(pdn) && /左|[Ll]eft/i.test(pdn)))   { L = v; hitCount++; }
                else if (/^[Rr]ight$|^右$|^右端$/.test(pdn) || (/切り抜き|[Cc]rop/i.test(pdn) && /右|[Rr]ight/i.test(pdn))) { R = v; hitCount++; }
                else if (/^[Tt]op$|^上$|^上端$/.test(pdn) || (/切り抜き|[Cc]rop/i.test(pdn) && /上|[Tt]op/i.test(pdn)))     { T = v; hitCount++; }
                else if (/^[Bb]ottom$|^下$|^下端$/.test(pdn) || (/切り抜き|[Cc]rop/i.test(pdn) && /下|[Bb]ottom/i.test(pdn))) { B = v; hitCount++; }
            }
            var isCropFx = /[Cc]rop|クロップ|切り抜き/.test(cdn);
            var hasNonZero = (L > 0 || R > 0 || T > 0 || B > 0);
            if (hitCount >= 2 && hasNonZero) {
                var maxV = Math.max(isNaN(L)?0:L, isNaN(R)?0:R, isNaN(T)?0:T, isNaN(B)?0:B);
                var sc = (maxV > 0 && maxV <= 1) ? 100 : 1;
                var cand = { leftPct: isNaN(L)?0:L*sc, rightPct: isNaN(R)?0:R*sc, topPct: isNaN(T)?0:T*sc, bottomPct: isNaN(B)?0:B*sc };
                if (isCropFx || cdn === 'Motion' || cdn === 'モーション') {
                    if (!cropResult || isCropFx) { cropResult = cand; }
                }
            }
        }
    } catch (e) {}
    return cropResult;
}

function _intersect(a, b) {
    return { left: Math.max(a.left,b.left), top: Math.max(a.top,b.top), right: Math.min(a.right,b.right), bottom: Math.min(a.bottom,b.bottom) };
}
```
